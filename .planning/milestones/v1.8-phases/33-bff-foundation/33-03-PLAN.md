Create `bff/src/session.js`. This module builds ONE pg.Pool for the app lifetime (do not create a new Pool per request or inside the store config twice), passes it to connect-pg-simple, and wraps it in express-session. Export both `pool` (for potential reuse by auth routes in Phase 34) and `sessionMiddleware` (consumed by index.js).

Implementation:

```javascript
'use strict';
const session = require('express-session');
const pgSession = require('connect-pg-simple')(session);
const { Pool } = require('pg');
const config = require('./config');

const pool = new Pool({
  host: config.db.host,
  port: config.db.port,
  database: config.db.database,
  user: config.db.user,
  password: config.db.password,
});

const store = new pgSession({
  pool,
  tableName: 'session',
  schemaName: config.db.schema,
  createTableIfMissing: true,
  pruneSessionInterval: 900,
});

const sessionMiddleware = session({
  store,
  name: 'bff.sid',
  secret: config.sessionSecret,
  resave: false,
  saveUninitialized: false,
  proxy: true,
  cookie: {
    httpOnly: true,
    secure: true,
    sameSite: 'lax',
    domain: '.pm.comsatel.com.pe',
    maxAge: 28800000,
  },
});

module.exports = { pool, sessionMiddleware };
```

Critical notes:
- `saveUninitialized: false` — do NOT create a session row for unauthenticated visitors. A session row is only written when something sets `req.session.someKey = value`. After a plain `GET /bff/health`, the session table should have 0 rows (health route does not touch the session).
- `resave: false` — connect-pg-simple implements the `touch()` method, so this is safe and correct.
- `proxy: true` — redundant with `app.set('trust proxy', 1)` but harmless; include for explicitness.
- The `pool` is shared — the same pool used for sessions can later be reused by Phase 34 auth queries.
