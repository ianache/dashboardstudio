    ```python
    APP_NAME = "ai-analyst"

    AGENT_INSTRUCTION = (

    def create_runner(model_str: str, deepseek_api_key: str | None = None) -> Runner:
        """Factory: constructs a fresh LlmAgent + Runner for the requested model.

        Uses api_key constructor param (not os.environ) to avoid race conditions
        in async context. session_service is reused across requests.
        """
        if model_str.startswith("deepseek/"):
            model = LiteLlm(
                model=model_str,
                api_key=deepseek_api_key or "",
                stream_options={"include_usage": True},
            )
        else:
            # Gemini — plain string, ADK resolves natively
            model = model_str

        agent = LlmAgent(
            name="bi_analyst",
            model=model,
            tools=[query_data, execute_skill],
            instruction=AGENT_INSTRUCTION,
        )
        return Runner(app_name=APP_NAME, agent=agent, session_service=session_service)
    ```

    CRITICAL details:
    - Copy AGENT_INSTRUCTION verbatim from existing file (the long multi-line string starting with "You are the 'BI Analyst'...")
    - Do NOT export `root_agent` or `runner` — only `create_runner`, `session_service`, `APP_NAME`
    - `stream_options={"include_usage": True}` is REQUIRED for DeepSeek usage metadata
    - Agent name uniqueness: If ADK raises ValueError on concurrent `name="bi_analyst"` instances, add `import uuid` and use `name=f"bi_analyst_{uuid.uuid4().hex[:8]}"` instead

    ```python
    PRICING = {
        "gemini-2.5-flash-lite":          {"input": 0.000000075, "output": 0.0000003},
        "deepseek/deepseek-v4-flash":     {"input": 0.00000027,  "output": 0.0000011},

        rates = PRICING.get(model_str, {"input": 0.0, "output": 0.0})
        return round(input_tokens * rates["input"] + output_tokens * rates["output"], 6)
    ```
